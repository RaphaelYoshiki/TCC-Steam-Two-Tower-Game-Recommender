FROM python:3.9.13-slim-bullseye

ENV PYTHONUNBUFFERED=1 \
    PIP_NO_CACHE_DIR=1 \
    TF_CPP_MIN_LOG_LEVEL=2

# Dependências de sistema
RUN apt-get update \
 && apt-get install -y --no-install-recommends \
    build-essential git curl ca-certificates \
    libglib2.0-0 libsm6 libxrender1 libxext6 libgomp1 \
 && rm -rf /var/lib/apt/lists/*

WORKDIR /app

# Copia requirements e instala dependências
COPY scripts/requirements.txt /app/requirements.txt
RUN python -m pip install --upgrade pip setuptools wheel \
 && pip install --no-cache-dir -r requirements.txt

# Cria diretórios necessários com permissões corretas
RUN mkdir -p /app/models /app/model_plots /app/temp /app/csv_files \
 && chmod 777 /app/models /app/model_plots /app/temp /app/csv_files

# Copia o projeto
COPY . /app

# Rodar como usuário não-root
RUN useradd -m appuser && chown -R appuser:appuser /app
USER appuser

CMD ["python", "scripts/two_tower_training_NOCHUNK.py"]
